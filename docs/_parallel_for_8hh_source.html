36 assert(workload == nitems || nitems == 0);

38 const std::ptrdiff_t chunkedWorkPerThread = std::max<std::ptrdiff_t>(std::llround(workPerThread/3.0), 1);

119 inline void parallel_foreach_impl(Manager & pool, [[maybe_unused]] std::ptrdiff_t nitems,
120 Iteration iter, Iteration end, Function && function, std::input_iterator_tag)
121 {
122 [[maybe_unused]] std::ptrdiff_t num_items = 0;
123 std::vector<std::future<void>> futures;
124 for (; iter != end; ++iter)
125 {
126 auto item = *iter;
127 futures.emplace_back(pool.enqueue([&function, &item](Integer id){function(id, item);}));
128 ++num_items;
129 }
130 assert(num_items == nitems || nitems == 0);
131 for (auto & future : futures) {future.get();}
